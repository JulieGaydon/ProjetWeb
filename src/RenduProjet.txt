Components :
- PagePrincipale
- Login

- Actualités
- Signin
- Actualités
- Message
- ListeMessage
- Utilisateur

Login : etat=false
    - connexion() :
        change l'etat de connexion en true
    - EventConnected() :
        permet d'afficher la bonne page selon le bouton clique (PagePrincipale, Mur de Tweet)

Logout : etat=true
    permet l'affichage du bouton deconnexion lorsqu'on est connecte

Signin : etat=false
    - inscription() :
        change etat de connexion a true
    - EventConnected() :
        affiche formulaire inscription, permet d'afficher la bonne page selon bouton clique  (PagePrincipale, Mur de Tweet)

PagePrincipale : etat=false
    - connection() :
        change etat du bouton connexion a true
    - inscription() :
        change etat du bouton inscription a true
    - EventConnected() :
        permet d'afficher la bonne page selon le bouton clique (Login ou Signin)

Utilisateur : etat=true

Message : etat=true

ListeMessage : etat=true

Actualites : etat=true
